import PasswordInputContainer from './PasswordInput.state.js';

## Basic
<PasswordInputContainer />

```jsx
<PasswordInput
  value={this.state.password}
  propName="password"
  onChange={(e, value) => this.setState({ password: value })}
  label="Password"
/>
```

## Validation Function

```jsx
parseValue = () => {
  const { value = '' } = this.props;

  return {
    lower: {
      completed: /[a-z]/.test(value),
      message: 'At Least one Lowercase Letter'
    },
    upper: {
      completed: /[A-Z]/.test(value),
      message: 'At Least one Uppercase Letter'
    },
    number: {
      completed: /[0-9]/.test(value),
      message: 'At Least one Number'
    },
    specialChar: {
      completed: /[!@#\\$%\\^&\\*]/.test(value),
      message: 'At Least one Special Character'
    },
    size: {
      completed: value.length >= 8,
      message: 'At Least 8 Characters long'
    }
  }
}
```

## Props

| Property    | Type             | Required | Description |
| :-------    | :---             | :------- | :---------- |
| value       | `string`/`array` | true     | Value of the Input |
| propName    | `string`         | false    | Property name |
| className   | `string`         | false    | className of input |
| label       | `string`         | false    | Label of input |
| onChange    | `function`       | true     | Function fired when input's `onChange` event triggers. Arguments: (e, value, propName).|
| onFocus     | `function`       | false    | Function fired when input's `onFocus` event triggers.|
| onBlur      | `function`       | false    | Function fired when input's `onBlur` event triggers.|
| placeholder | `string`         | false    | Placeholder of input |
| disabled    | `boolean`        | false    | Disabled of input |
